apiVersion: apps/v1
kind: Deployment
metadata:
  name: mongodb
  namespace: easyshop  # Add your namespace if you're using one
spec:
  replicas: 1
  selector:
    matchLabels:
      app: mongodb
  template:
    metadata:
      labels:
        app: mongodb
    spec:
      containers:
      - name: mongodb
        image: mongo:6.0
        ports:
          - containerPort: 27017
        volumeMounts:
          - name: mongo-storage
            mountPath: /data/db  # This is where MongoDB stores its data
      volumes:
        - name: mongo-storage
          persistentVolumeClaim:
            claimName: mongo-pvc  # This should match your PVC name
